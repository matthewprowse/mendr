## Audit Method

**Tools used:**
- `find` — directory and file enumeration
- `grep -r` — import discovery across codebase
- Manual inspection of suspect files

**Commands executed:**
```bash
# Count all TS/TSX files in scope
find /Users/matthewprowse/Documents/Development/Personal/Home\ Services/app/src/lib \
  -type f \( -name "*.ts" -o -name "*.tsx" \) | wc -l
# Result: 133 files

# Find all files with zero imports
for file in $(find src/lib -type f \( -name "*.ts" -o -name "*.tsx" \) | grep -v "test.ts" | grep -v "__tests__"); do
    import_path="@/lib/${file#src/lib/}"
    count=$(grep -r "from ['\"]$import_path['\"]" . 2>/dev/null | wc -l)
    [ "$count" -eq 0 ] && echo "$file — 0 imports"
done

# Check for forbidden patterns
grep -r "catch.*: any" src/lib --include="*.ts" --include="*.tsx"
grep -r "console\.log" src/lib --include="*.ts" --include="*.tsx"
grep -r "@ts-ignore\|@ts-nocheck" src/lib --include="*.ts" --include="*.tsx"
grep -r "// TODO\|// FIXME\|// XXX" src/lib --include="*.ts" --include="*.tsx"

# Verify import graph (lib should not import from app)
grep -r "from ['\"]@/app/" src/lib --include="*.ts" --include="*.tsx"

# Line counts (identify large files)
find src/lib -type f \( -name "*.ts" -o -name "*.tsx" \) ! -path "*/__tests__/*" \
  -exec wc -l {} + | sort -rn | head -30
```
